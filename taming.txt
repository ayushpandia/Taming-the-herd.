#include <bits/stdc++.h>
using namespace std;
int PI = (int) 1e8;
int main ()
{
  ios_base::sync_with_stdio (false);
  cin.tie (NULL);
  ifstream cin("taming.in");
  ofstream cout("taming.out");
  int n;cin >> n;
  int a[n];
  for (auto & aa:a)cin >> aa;
  int dp[n + 1][n + 1][n + 1];
  for(int i = 0; i <= n; i++)
      for(int j=0;j<=n;j++)
        for (int k=0;k<=n;k++)dp[i][j][k]=PI;
 
  if(a[0]==0)dp[0][1][0]=0;
  else dp[0][1][0]=1;
  int ans;
  for(int i=1;i<n;i++)
  {
      for(int j=1;j<=n;j++)
      {
          for(int k=0;k<n;k++)
          {
               if(k==0)
               {
                   ans=PI;
                   for(int next=0;next<n;next++)
                        ans=min(ans,dp[i-1][j-1][next]);
                   dp[i][j][k]=ans+((a[i]!=k)?1:0);
               }
               else
               {
                   dp[i][j][k]=((a[i]!=k)?1:0) + dp[i-1][j][k-1];
               }
          }
      }
  }
  
    for(int i=0;i<n;i++)
    {
        int ans=PI;
        for(int start=0;start<n;start++)
            ans=min(ans,dp[n-1][i+1][start]);
        cout<<ans<<"\n";
    }
    return 0;
}
